        /**
         * Generates the HTML for a single tiered product recommendation card.
         */
        function renderTieredRecommendationCard(rec) {
            const amazonSearchQuery = encodeURIComponent(rec.name);
            const amazonUrl = `https://www.amazon.in/s?k=${amazonSearchQuery}`; // Updated to Amazon India (.in)

            let headerColor = '';
            let buttonColor = 'bg-indigo-600 hover:bg-indigo-700';

            switch (rec.type) {
                case 'budget': headerColor = 'text-green-400'; break;
                case 'mid-range': headerColor = 'text-cyan-400'; break;
                case 'high-end': headerColor = 'text-red-400'; break;
                default: headerColor = 'text-gray-400';
            }

            return `
                <div class="p-5 bg-gray-800 rounded-xl border-t-8 border-indigo-500/50 shadow-xl flex flex-col transition duration-300 hover:shadow-2xl hover:shadow-indigo-500/40">
                    <h4 class="text-xl font-bold ${headerColor} mb-1">${rec.tier}</h4>
                    <p class="text-2xl font-extrabold text-white mb-2">${rec.name}</p>
                    <p class="text-sm text-gray-400 flex-grow mb-4">${rec.description}</p>
                    <div class="flex items-center justify-between mt-auto mb-4">
                        <span class="text-xl font-bold text-yellow-400">${rec.price}</span>
                    </div>
                    <a href="${amazonUrl}" target="_blank" rel="noopener noreferrer" 
                       class="mt-auto inline-flex items-center justify-center px-4 py-2 ${buttonColor} text-white font-medium text-base rounded-lg transition duration-150 transform hover:scale-[1.02] shadow-lg">

                        <i class="fa-solid fa-magnifying-glass ml-2"></i>
